    <script>
        // Data State
        let items = JSON.parse(localStorage.getItem('zen_data')) || [];

        // Digital Clock
        setInterval(() => {
            document.getElementById('digital-clock').innerText = new Date().toLocaleTimeString('id-ID');
        }, 1000);

        function render() {
            const list = document.getElementById('item-list');
            const balanceEl = document.getElementById('balance');
            list.innerHTML = '';
            
            let totalBalance = 0;

            if (items.length === 0) {
                list.innerHTML = '<div class="text-center py-20 text-slate-400 italic font-medium">Belum ada data tersedia.</div>';
            }

            items.forEach((item, index) => {
                let colorClass = 'bg-blue-50 text-blue-600';
                let icon = '📌';
                let sign = '';

                if (item.type === 'income') {
                    colorClass = 'bg-emerald-50 text-emerald-600';
                    icon = '💰';
                    sign = '+';
                    totalBalance += parseInt(item.val);
                } else if (item.type === 'expense') {
                    colorClass = 'bg-rose-50 text-rose-600';
                    icon = '📉';
                    sign = '-';
                    totalBalance -= parseInt(item.val);
                }

                const div = document.createElement('div');
                div.className = `flex items-center justify-between p-5 bg-white rounded-2xl shadow-sm border border-slate-100 card-hover transition-all animate-in fade-in slide-in-from-right-4 duration-300`;
                div.innerHTML = `
                    <div class="flex items-center gap-4">
                        <div class="w-12 h-12 ${colorClass} rounded-2xl flex items-center justify-center text-xl shadow-inner">
                            ${icon}
                        </div>
                        <div>
                            <h4 class="font-bold text-slate-800">${item.desc}</h4>
                            <p class="text-[10px] text-slate-400 font-bold uppercase tracking-wider">${item.type}</p>
                        </div>
                    </div>
                    <div class="flex items-center gap-6">
                        <span class="font-black ${item.type === 'income' ? 'text-emerald-500' : (item.type === 'expense' ? 'text-rose-500' : 'text-slate-700')}">
                            ${item.type === 'todo' ? 'Lvl ' + item.val : 'Rp ' + sign + parseInt(item.val).toLocaleString()}
                        </span>
                        <button onclick="deleteItem(${index})" class="p-2 text-slate-200 hover:text-rose-500 transition-colors">
                            <svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M3 6h18"/><path d="M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6"/><path d="M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2"/></svg>
                        </button>
                    </div>
                `;
                list.appendChild(div);
            });

            balanceEl.innerText = 'Rp ' + totalBalance.toLocaleString();
            localStorage.setItem('zen_data', JSON.stringify(items));
        }

        function addItem() {
            const type = document.getElementById('type').value;
            const desc = document.getElementById('desc').value;
            const val = document.getElementById('val').value;

            if (!desc || !val) {
                alert('Mohon isi semua data!');
                return;
            }

            items.unshift({ type, desc, val });
            document.getElementById('desc').value = '';
            document.getElementById('val').value = '';
            render();
        }

        function deleteItem(index) {
            items.splice(index, 1);
            render();
        }

        function clearAll() {
            if(confirm('Hapus semua data?')) {
                items = [];
                render();
            }
        }

        // Initial render
        render();
    </script>
